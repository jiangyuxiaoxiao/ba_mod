# Generate them from a generic template

every_possible_country = {
	limit = {
		NOT = {
			TAG = HIL
			TAG = KRO
			TAG = CAO
			TAG = CAP
			TAG = CAB
			TAG = CAH
		}
		has_idea = BARE_Global_Trusteeship_Area
		
	}
	generate_character = {
		token_base = generic_Global_Trusteeship_Area_Leader

		name = Global_Trusteeship_Area_Leader_name
		portraits = {
			civilian = {
				large = GFX_portrait_Global_Trusteeship_Area_Leader
			}
		}
		country_leader = {
			ideology = no_ideology_type
			expire = "9999.1.1.1"
			id = -1
		}
	}
}
