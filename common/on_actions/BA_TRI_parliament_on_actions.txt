on_actions = {
	on_startup = {
		effect = {
			TRI = {
				add_ideas = {
					TRI_center_campus_region
					TRI_western_campus_region
					TRI_northern_campus_region
					TRI_eastern_campus_region
					TRI_southern_campus_region
					TRI_industrial_park_region
					TRI_living_area_region
					TRI_business_district_region
				}
				set_variable = { TRI_selected_region = token:TRI_center_campus_region }
				set_variable = { TRI_region_target_association = token:TRI_parliament_association_3 }
				set_variable = { TRI_region_target_work = token:TRI_region_military }
				set_variable = { TRI_regions_adm_mil_select_button = 2 }
				set_variable = { TRI_regions_adm_cul_select_button = 0 }
				set_variable = { TRI_regions_adm_soc_select_button = 0 }
				set_variable = { TRI_teaparty_adm_limit = 240 }
				TRI_region_generate_statistics = yes
				TRI_parliament_house_election = yes
				add_dynamic_modifier = {
					modifier = TRI_region_modifier
				}
				every_controlled_state = {
					limit = { has_variable = state_region }
					add_dynamic_modifier = {
						modifier = TRI_state_region_stability_modifier
					}
				}
			}
		}
	}
	on_daily_TRI = {
		effect = {
			if = { #花子执政效果
				limit = { 
					has_country_leader = { character = TRI_hanako }
					has_variable = TRI_parliament_focus
					has_completed_focus = TRI_across_the_border
				}
				set_variable = { support@TRI_parliament_association_4 = 0 }
				set_variable = { centrist@TRI_parliament_association_4 = 0 }
				set_variable = { opposition@TRI_parliament_association_4 = 0 }
				if = {
					limit = { check_variable = { BA_TRI_player_position = token:position_support } }
					set_variable = { support@TRI_parliament_association_4 = 20 }
				}
				else_if = {
					limit = { check_variable = { BA_TRI_player_position = token:position_centrist } }
					set_variable = { centrist@TRI_parliament_association_4 = 20 }
				}
				else = {
					set_variable = { opposition@TRI_parliament_association_4 = 20 }
				}
			}

			if = {
				limit = { check_variable = { TRI_political_focus_cd > 0 } }
				subtract_from_variable = { TRI_political_focus_cd = 1 }
			}
		}
	}
	on_monthly_TRI = {
		effect = {
			TRI_region_generate_state_modifiers = yes
			TRI_region_generate_statistics = yes
			TRI_BOP_init_parliament = yes
			TRI_init_senate_house = yes
		}
	}
}
