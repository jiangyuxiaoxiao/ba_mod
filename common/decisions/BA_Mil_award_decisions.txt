        days_remove = 45
        days_re_enable = 45
        modifier = {
            consumer_goods_factor = 0.1
        }
        remove_effect = {
            add_days_mission_timeout = {
                mission = MIL_award_imaginary_number_network_new_invention
                days = -45
            }
        }
    }
    MIL_award_imaginary_number_network_more_people = {
        days_remove = 45
        days_re_enable = 45
        modifier = {
            research_speed_factor = -0.05
        }
        remove_effect = {
            add_days_mission_timeout = {
                mission = MIL_award_imaginary_number_network_new_invention
                days = -45
            }
        }
    }
    MIL_award_imaginary_number_network_fix = {
        
    }
    MIL_award_imaginary_number_network_update = {
        cost = 25
        days_remove = 45
        days_re_enable = 45
        complete_effect = {
            add_days_mission_timeout = {
                mission = MIL_award_imaginary_number_network_new_invention
                days = 15
            }
        }
        remove_effect = {
            add_to_variable = {
                MIL_award_imaginary_number_network_min = 5
            }
            add_to_variable = {
                MIL_award_imaginary_number_network_max = 5
            }
        }
    }
    MIL_award_imaginary_number_network_new_invention = {
        allowed = { tag = MIL }
        available = {
            always = no
        }
        timeout_effect = {
            randomize_variable = {
                var = MIL_award_imaginary_number_network_prob
                distribution = uniform
                min = MIL_award_imaginary_number_network_min
                max = MIL_award_imaginary_number_network_max
            }
            if = {
                limit = { check_variable = { MIL_award_imaginary_number_network_prob > 85 } }
            }
            else_if = {
                limit = { check_variable = { MIL_award_imaginary_number_network_prob > 40 } }
            }
            else_if = {
                limit = { check_variable = { MIL_award_imaginary_number_network_prob > 10 } }
            }
            else = {

            }
        }

        is_good = yes
    }
