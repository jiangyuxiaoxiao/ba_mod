			set_variable = { enable@TRI_koharu_inv3 = 1 }

    	name = TRI_events.210.a

			localization_key = TRI_investigation_safe_tt
    		SAFE = TRI_safe
			NAME = TRI_koharu_name
		}
		set_variable = { permission@TRI_koharu = 1 }
		add_to_variable = { TRI_permitted_hoshubu_member = 1 }
		TRI_paranoia_decrease_10 = yes
		if = {
			limit = { 
				NOT = { has_country_flag = TRI_investigate_open }
			}
			set_country_flag = TRI_investigate_open
		}
    	ai_chance = {
    		base = 1
    	}
    }
}
